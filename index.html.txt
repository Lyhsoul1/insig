<div id="totalBonus">Bônus total: 0</div>

<script>
// Aqui você pode editar os addons e seus bônus
const addons = [
  { nome: "Dragon Helmet", bonus: 5 },
  { nome: "Golden Armor", bonus: 8 },
  { nome: "Shadow Boots", bonus: 3 },
  { nome: "Phoenix Shield", bonus: 7 },
  { nome: "Crystal Sword", bonus: 6 }
];

// Função que monta a lista de checkboxes
function montarLista() {
  const container = document.getElementById('addonsList');
  container.innerHTML = ''; // limpa antes

  addons.forEach((addon, i) => {
    const label = document.createElement('label');
    label.innerHTML = `
      <input type="checkbox" value="${i}" />
      ${addon.nome} (+${addon.bonus})
    `;
    container.appendChild(label);
  });
}

// Função que calcula e mostra o total de bônus
function calcularBonus() {
  const checkboxes = document.querySelectorAll('#addonsList input[type="checkbox"]');
  let total = 0;
  checkboxes.forEach(cb => {
    if (cb.checked) {
      const idx = parseInt(cb.value);
      total += addons[idx].bonus;
    }
  });
  document.getElementById('totalBonus').textContent = `Bônus total: ${total}`;
}

// Monta lista ao carregar a página
montarLista();

// Adiciona listener para recalcular quando clicar nas checkboxes
document.getElementById('addonsList').addEventListener('change', calcularBonus);

</script>
